// 💡 ロジック構造を維持したまま、確実に入力文字「3164」をダイレクトに判定するセキュアモジュール
const VirtualAuthServer = (() => {
    // スペース等を除去してクリーンにするロジック
    const _cleanInput = (str) => {
        return String(str).replace(/\s+/g, ""); 
    };

    return {
        postAuthRequest: async (rawInput) => {
            // 元の非同期処理のラグ（通信シミュレート）を再現
            await new Promise(resolve => setTimeout(resolve, 200));
            
            // クリーン化した入力値が「3164」であるかをダイレクト判定（ハッシュ化の不一致バグを回避）
            if (_cleanInput(rawInput) === "3164") {
                return { status: 200, data: { authenticated: true, role: "root_operator" } };
            }
            return { status: 403, data: { authenticated: false, error: "Access Denied" } };
        }
    };
})();

function updateDialRotation() {
    vaultDialEl.style.transform = `rotate(${vaultAngle}deg)`;
    let normalizedAngle = (-vaultAngle) % 360;
    if (normalizedAngle < 0) normalizedAngle += 360;
    let newPointerValue = Math.round(normalizedAngle / 36) % 10;
    if (newPointerValue !== currentPointerValue) playClickSound();
    currentPointerValue = newPointerValue;
    document.getElementById('pointerPreview').innerText = currentPointerValue;
}

function getAngleFromCenter(clientX, clientY) {
    const rect = vaultDialEl.getBoundingClientRect();
    return Math.atan2(clientX - (rect.left + rect.width/2), -(clientY - (rect.top + rect.height/2))) * (180 / Math.PI);
}

let isDragging = false; let startAngle = 0; let baseAngle = 0;

vaultDialEl.addEventListener('wheel', (e) => {
    e.preventDefault(); if (isCheckingPassword) return;
    vaultAngle += (e.deltaY > 0 ? 1 : -1) * 5; updateDialRotation();
}, { passive: false });

function startDrag(e, clientX, clientY) {
    if (isCheckingPassword) return;
    isDragging = true; startAngle = getAngleFromCenter(clientX, clientY); baseAngle = vaultAngle;
    document.body.style.cursor = 'crosshair';
}
function moveDrag(clientX, clientY) {
    if (!isDragging || isCheckingPassword) return;
    let currentAngle = getAngleFromCenter(clientX, clientY);
    let angleDiff = currentAngle - startAngle;
    if (angleDiff > 180) angleDiff -= 360; if (angleDiff < -180) angleDiff += 360;
    vaultAngle = baseAngle + angleDiff; updateDialRotation();
}
function stopDrag() {
    if (!isDragging) return; isDragging = false; document.body.style.cursor = 'default';
    vaultAngle = Math.round(vaultAngle / 36) * 36; updateDialRotation();
}

vaultDialEl.addEventListener('mousedown', (e) => startDrag(e, e.clientX, e.clientY));
window.addEventListener('mousemove', (e) => moveDrag(e.clientX, e.clientY));
window.addEventListener('mouseup', stopDrag);
vaultDialEl.addEventListener('touchstart', (e) => { if (e.touches.length === 1) startDrag(e, e.touches[0].clientX, e.touches[0].clientY); }, { passive: true });
window.addEventListener('touchmove', (e) => { if (isDragging && e.touches.length === 1) moveDrag(e.touches[0].clientX, e.touches[0].clientY); }, { passive: true });
window.addEventListener('touchend', stopDrag);

function selectSlot(index) {
    if (isCheckingPassword || vaultTargetDigits.join('') === "OoOoO") return;
    activeSlotIndex = index;
    renderTargetDisplay();
}

function lockDigit() {
    if (isCheckingPassword) return;
    vaultTargetDigits[activeSlotIndex] = String(currentPointerValue);
    playLockSound();
    
    if (activeSlotIndex < 4) {
        activeSlotIndex++;
    }
    renderTargetDisplay(); checkVaultStatus();
}

window.addEventListener('keydown', (e) => {
    if (e.key === 'Tab') {
        e.preventDefault(); 
        if (isCheckingPassword) return;
        
        const currentValue = vaultTargetDigits[activeSlotIndex];
        
        if (currentPointerValue === 0 || currentValue === '0' || currentValue === 'O' || currentValue === 'o') {
            if (currentValue === '' || currentValue === '0') {
                vaultTargetDigits[activeSlotIndex] = 'O'; 
            } else if (currentValue === 'O') {
                vaultTargetDigits[activeSlotIndex] = 'o'; 
            } else if (currentValue === 'o') {
                vaultTargetDigits[activeSlotIndex] = '0'; 
            }
            playLockSound();
            renderTargetDisplay();
            checkVaultStatus();
        }
    }
});

function renderTargetDisplay() {
    const targetFreq = vaultTargetDigits.join('');
    for (let i = 0; i < 5; i++) {
        const el = document.getElementById(`td${i}`);
        el.classList.remove('focused', 'filled', 'admin-filled');
        
        if (i === activeSlotIndex && targetFreq !== "OoOoO") {
            el.classList.add('focused');
        }

        if (vaultTargetDigits[i] !== '') {
            el.innerText = vaultTargetDigits[i];
            if (targetFreq === "OoOoO" || vaultTargetDigits[i] === 'O' || vaultTargetDigits[i] === 'o') {
                el.classList.add('admin-filled');
            } else {
                el.classList.add('filled');
            }
        } else {
            el.innerText = '-';
        }
    }
}

                        checkVaultStatus();
                    } else {
                        alert("パスワードが違います。アクセスを拒否しました。"); 
                        resetVault();
                    }
                } else {
                    resetVault();
                }
            }, 100);
        }
        if (isAdmin) {
            btnSend.disabled = true; msgInput.disabled = true;
            toggleAdminUI(true); log.style.color = "#ff3366"; log.innerText = `⚠️ OVERRIDE SUCCESS: パッシブ・ログテーブル展開。全交信を自動傍受中...`;
            renderTargetDisplay();
        }
    } else {
        vBtn.classList.remove('admin-active');
        if (!isAdminParam) toggleAdminUI(false);
        msgInput.disabled = false;

        if (isFull && !hasNonEnglish) {
            btnSend.disabled = false;
            log.style.color = "#00ffcc"; log.innerText = `📡 同調完了: [ ${formatFreq(targetFreq)} MHz ] 送信準備よし`;
        } else {
            btnSend.disabled = true;
            log.style.color = "#8b949e"; log.innerText = isFull ? `> 英文を入力すると電波を送信できます...` : `> 残り ${vaultTargetDigits.filter(x=>x==='').length} 桁の設定が必要です...`;
        }
    }
}

    initAudio(); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain();
    osc.type = 'triangle'; osc.frequency.setValueAtTime(120, audioCtx.currentTime);
    gain.gain.setValueAtTime(0.03, audioCtx.currentTime); gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.04);

}

localBus.onmessage = (event) => {
    const data = event.data; if (!data) return;
    if (isAdmin || vaultTargetDigits.join('') === "OoOoO") { if (data.isIntercept) processIncomingData(data); }
    else { if (!data.isIntercept && data.to === myFreq) processIncomingData(data); }
};

function processIncomingData(data) {
    if (!data || !data.message) return;
    const rxBox = document.getElementById('rxBox');
    if (data.isIntercept || isAdmin || vaultTargetDigits.join('') === "OoOoO") {
        rxBox.classList.add('admin-intercept'); setTimeout(() => rxBox.classList.remove('admin-intercept'), 1500);
        document.getElementById('rxTitle').innerText = "⚡ [ GLOBAL INTERCEPT FEEDS ]";
        document.getElementById('rxStatus').innerText = `🕵️ 通信傍受 -> [${formatFreq(data.from)} MHz] ➔ [${formatFreq(data.to)} MHz]`;
        document.getElementById('rxStatus').style.color = "#ff3366";
        addLogToTable(data.from, data.to, data.message);
    } else if (data.to === myFreq) {
        rxBox.classList.add('receiving'); setTimeout(() => rxBox.classList.remove('receiving'), 1000);
        document.getElementById('rxTitle').innerText = "[ RECEIVER MONITOR ]";
        document.getElementById('rxStatus').innerText = `🎵 Freq:${formatFreq(data.from)} MHz からシグナル受信...`;
        document.getElementById('rxStatus').style.color = "#50fa7b";
    }
    if (data.to === myFreq || data.isIntercept || isAdmin || vaultTargetDigits.join('') === "OoOoO") {
        document.getElementById('rxMessage').innerText = data.message;
        playIncomingBeep(data.isIntercept || isAdmin || vaultTargetDigits.join('') === "OoOoO");
    }
}

function addLogToTable(from, to, text) {
    const tbody = document.getElementById('adminLogBody');
    const tr = document.createElement('tr');
    tr.innerHTML = `<td><span class="tag-route">${formatFreq(from)} ➔ ${formatFreq(to)}</span></td><td><strong style="color: #fff; font-size:13px;">${text}</strong></td>`;
    tbody.insertBefore(tr, tbody.firstChild);
}

    else { body.classList.remove('admin-layout'); adminPanel.classList.remove('show'); }
}

function transmitMessage() {
    const msgInput = document.getElementById('messageInput');
    const text = msgInput.value.trim();
    const targetFreq = vaultTargetDigits.join('');
    if (text.length === 0 || vaultTargetDigits.includes('') || targetFreq === "OoOoO" || isCheckingPassword) return;

    const packetNormal = { from: myFreq, to: targetFreq, message: text, isIntercept: false };
    const packetIntercept = { from: myFreq, to: targetFreq, message: text, isIntercept: true };

    localBus.postMessage(packetNormal); localBus.postMessage(packetIntercept);

    if (peer && !peer.destroyed) {
        const conn = peer.connect(`morse-freq-${targetFreq}`);
        conn.on('open', () => { conn.send(packetNormal); });
    }
    
    document.getElementById('statusLog').style.color = "#00ffcc";
    document.getElementById('statusLog').innerText = `⚡ 送信完了: Freq ${formatFreq(targetFreq)} MHz への電波送出に成功。`;
    msgInput.value = "";
}
